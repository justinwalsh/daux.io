	    signal QT: std_logic; -- Q(t)
	begin
	    process(clock, reset) is
	        subtype RS is std_logic_vector (1 downto 0);
	    begin
	        if reset = '0' then

	        else
	            if rising_edge(C) then
	                if not (R'stable(T) and S'stable(T)) then

	                else

	                        when "11" => QT <= 'X';

	        end if;
	    end process;
